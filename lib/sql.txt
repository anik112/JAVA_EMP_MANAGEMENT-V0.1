CREATE TABLE PAYROLL.TB_COMPANY_INFO
   (	
    COMID int(11) NOT NULL AUTO_INCREMENT, 
	COMPANY VARCHAR(255) NOT NULL, 
	ADDRESS VARCHAR(255), 
	ADDRESS1 VARCHAR(255), 
	UNIT VARCHAR(255), 
	HRSIGN_WORKER BLOB, 
	HRSIGN_STAFF BLOB, 
	COMLOGO BLOB, 
	ISOLOGO BLOB, 
	PROCESSBY VARCHAR(30), 
	PROCESSDATE DATE, 
	COMPANY_BANGLA VARCHAR(250), 
	HR_HEAD_NAME VARCHAR(200), 
	SHIFT VARCHAR(5), 
	PHONE_NO VARCHAR(35),
    PRIMARY KEY(COMID)
   )ENGINE=InnoDB;


CREATE TABLE PAYROLL.TB_DEPARTMENT_INFO
   (	
    DEPTID INT(11) NOT NULL AUTO_INCREMENT,
    COMID INT(11) NOT NULL,
	DEPT_NAME VARCHAR(100) NOT NULL, 
	DEPT_HEAD VARCHAR(100), 
	DEPT_HEAD_NAME VARCHAR(255), 
	DEPT_ADDRESS VARCHAR(100), 
	WEB_ADDRESS VARCHAR(100), 
	PHONE_OFF INT(15), 
	PROCESSBY VARCHAR(25), 
	PROCESSDATE DATE, 
	DEPT_NAME_BANGLA VARCHAR(100),
    PRIMARY KEY(DEPTID),
    FOREIGN KEY(COMID) REFERENCES PAYROLL.tb_company_info(COMID) ON DELETE CASCADE
   )ENGINE=InnoDB;
